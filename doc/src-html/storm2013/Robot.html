001    package storm2013;
002    
003    
004    import edu.wpi.first.wpilibj.IterativeRobot;
005    import edu.wpi.first.wpilibj.command.Command;
006    import edu.wpi.first.wpilibj.command.Scheduler;
007    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
008    import edu.wpi.first.wpilibj.smartdashboard.SendableChooser;
009    import storm2013.commands.Autonomous;
010    import storm2013.commands.Teleop;
011    import storm2013.subsystems.DriveTrain;
012    
013    /**
014     * The VM is configured to automatically run this class, and to call the
015     * functions corresponding to each mode, as described in the IterativeRobot
016     * documentation. If you change the name of this class or the package after
017     * creating this project, you must also update the manifest file in the resource
018     * directory.
019     */
020    public class Robot extends IterativeRobot {
021        public static OI oi;
022        public static DriveTrain driveTrain;
023    
024        Command   teleop;
025        String[]  autonomiceNames;
026        Command[] autonomice;
027        SendableChooser chooser = new SendableChooser();
028        Command   autonomouse;
029    
030        public void robotInit() {
031            oi = new OI();
032            driveTrain = new DriveTrain();
033            
034            teleop = new Teleop();
035            autonomice = new Command[]{ new Autonomous() };
036            for(int i=0;i<autonomice.length;++i) {
037                chooser.addObject(autonomiceNames[i],autonomice[i]);
038            }
039            
040        }
041    
042        public void autonomousInit() {
043            autonomouse = (Command)chooser.getSelected();
044        }
045    
046        /**
047         * This function is called periodically during autonomous
048         */
049        public void autonomousPeriodic() {
050            Scheduler.getInstance().run();
051        }
052    
053        public void teleopInit() {
054            autonomouse.cancel();
055        }
056    
057        /**
058         * This function is called periodically during operator control
059         */
060        public void teleopPeriodic() {
061            Scheduler.getInstance().run();
062        }
063        
064        /**
065         * This function is called periodically during test mode
066         */
067        public void testPeriodic() {
068            LiveWindow.run();
069        }
070    }
