005    package storm2013.commands;
006    
007    import edu.wpi.first.wpilibj.DigitalInput;
008    import edu.wpi.first.wpilibj.Victor;
009    import edu.wpi.first.wpilibj.command.Command;
010    import storm2013.RobotMap;
011    
012    /**
013     * Starts the motor for the thingy (so technical) that moves the ball to the firing wheel. 
014     * @author evan1026
015     */
016    public class StartBallFiringMotor extends Command {
017    
018        private Victor transferToWheelMotor;
019        private DigitalInput ballReadySensor;
020        
021        /**
022         * Makes a new one of these.
023         */
024        public StartBallFiringMotor(){
025            transferToWheelMotor = new Victor(RobotMap.PORT_MOTOR_BALL_FIRE_BELT);
026            ballReadySensor = new DigitalInput(RobotMap.PORT_BREAKBEAM_BALL_READY);
027        }
028        
029        /**
030         * {@inheritDoc}
031         */
032        protected void initialize() {
033            transferToWheelMotor.set(-1);
034        }
035    
036        /**
037         * {@inheritDoc}
038         */
039        protected void execute() {
040        }
041    
042        /**
043         * {@inheritDoc}
044         */
045        protected boolean isFinished() {
046            return ballReadySensor.get();
047        }
048    
049        /**
050         * {@inheritDoc}
051         */
052        protected void end() {
053        }
054    
055        /**
056         * {@inheritDoc}
057         */
058        protected void interrupted() {
059            end();
060        }
061    }
