001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    import storm2013.commands.LEDcommands.SetModeFiring;
006    import storm2013.subsystems.Tomahawk;
007    
008    /**
009     * Spins the tomahawk 1 revolution; sets the LED color as it does. To spin
010     * exactly one revolution, it runs until it meets the falling edge of
011     * tomahawk.isForward().
012     */
013    public class SpinTomahawk extends Command {
014        private final Command lightCommand = new SetModeFiring();
015        private final boolean _forward;
016        private boolean _hasChanged;
017        
018        public SpinTomahawk(boolean forward){
019            requires(Robot.tomahawk);
020            _forward = forward;
021        }
022        
023        protected void initialize() {
024            Robot.tomahawk.move(_forward);
025            _hasChanged = false;
026            lightCommand.start();
027        }
028        
029        protected void execute() {
030            if (Robot.tomahawk.isForward()) {
031                _hasChanged = true;
032            }
033        }
034        protected boolean isFinished() {
035            return _hasChanged && !Robot.tomahawk.isForward();
036        }
037        protected void end() {
038            Robot.tomahawk.stop();
039            lightCommand.cancel();
040        }
041        protected void interrupted() {
042            end();
043        }
044    }
