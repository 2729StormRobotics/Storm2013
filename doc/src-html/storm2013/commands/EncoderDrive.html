001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    import storm2013.subsystems.DriveTrain;
006    
007    /** Drives straight forward/back until a desired encoder value is reached. */
008    public class EncoderDrive extends Command {
009        private double _goal;
010        private double _dist;
011        private double _driveSpeed;
012        private DriveTrain _driveTrain = Robot.driveTrain;
013        
014        /**
015         * @param goal       Target encoder value
016         * @param driveSpeed Signal sent to the motors
017         */
018        public EncoderDrive(double goal, double driveSpeed){ 
019            
020            _goal = goal;
021            _driveSpeed = driveSpeed;
022            requires(Robot.driveTrain);
023        }
024        
025        protected void initialize() {
026           Robot.driveTrain.clearEncoder();
027        }
028        protected void execute() {
029            double left  = Robot.driveTrain.getLeftDistance(),
030                   right = Robot.driveTrain.getRightDistance();
031            
032            if(Math.abs(_goal-left) > Math.abs(_goal-right)) {
033                _dist = left;
034            }
035            else {
036                _dist = right;
037            }
038            
039            
040            if (_goal > 0) {
041                _driveTrain.tankDrive(_driveSpeed, _driveSpeed);
042            }
043            else {
044                _driveTrain.tankDrive(-_driveSpeed, -_driveSpeed);
045            }
046        }
047        protected boolean isFinished() {
048            if(_goal < 0) {
049                return _dist <= _goal;
050            } 
051            else {
052                return _dist >= _goal;
053            }
054        }
055        protected void end() {
056            Robot.driveTrain.tankDrive(0, 0);
057        }
058        protected void interrupted() {
059            end();
060        }
061    }
