010    public class DonaldDance extends CommandGroup {
011        class Forward extends Drive {
012            Forward() {
013                super(0.7,-0.5);
014            }
015        }
016        class Backward extends Drive {
017            Backward() {
018                super(-0.7,0.5);
019            }
020        }
021        class Left extends Drive {
022            Left() {
023                super(0,-1);
024            }
025        }
026        class Right extends Drive {
027            Right() {
028                super(0,1);
029            }
030        }
031        class Wait extends DoNothing {}
032        
033        public DonaldDance() {
034            addSequential(new Forward(),2);
035            addSequential(new Wait(),.5);
036            addSequential(new Right(),.8);
037            addSequential(new Wait(),.5);
038            addSequential(new Left(),1.6);
039            addSequential(new Wait(),.5);
