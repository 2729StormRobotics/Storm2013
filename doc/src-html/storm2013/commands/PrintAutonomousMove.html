001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /** Prints out code to be used in a CommandGroup for autonomous movement. */
007    public class PrintAutonomousMove extends Command {
008        private double _speed;
009        private double _waitTime;
010        
011        /**
012         * Constructor and such.
013         * @param speed    The speed to use for output.
014         * @param waitTime The time to wait between actions.
015         */
016        public PrintAutonomousMove(double speed,double waitTime){
017            _speed = speed;
018            _waitTime = waitTime;
019        }
020        
021        protected void initialize() {
022            String out = "addSequential(new ";
023            double left = Robot.driveTrain.getLeftDistance();
024            double right = Robot.driveTrain.getRightDistance();
025            double turn = Robot.driveTrain.getGyroAngle();
026            
027            if ((left < 0) != (right < 0)){
028                out += "GyroTurn(";
029                out += turn;
030            }
031            else {
032                out += "EncoderDrive(";
033                out += (right < 0) ? "-" : "";
034                out += (Math.min(Math.abs(left), Math.abs(right)));
035            }
036            
037            out += ", " + _speed + "));";
038            System.out.println(out);
039            
040            System.out.println("addSequential(new DoNothing()," + _waitTime + ");");
041            
042            Robot.driveTrain.clearEncoder();
043        }
044        protected void execute() {}
045        protected boolean isFinished() {
046            return true;
047        }
048        protected void end() {}
049        protected void interrupted() {
050            end();
051        }
052    }
