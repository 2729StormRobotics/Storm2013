001    package storm2013.commands.LEDcommands;
002    
003    import storm2013.Robot;
004    
005    /** Flash one color on and off (or solid on if period == 0). */
006    public class FlashOnAndOff extends Flash {
007        public FlashOnAndOff(Color color){
008            this(color, Flash.DEFAULT_PERIOD);
009        }
010        public FlashOnAndOff(Color color, double period){
011            super(new Color[] {color, Color.OFF}, period);
012        }
013        
014        public void execute(){
015            if (_period > 0){
016                super.execute();
017            }
018            else{
019                Robot.ledStrip.setColor(_colors[0].getRed(), _colors[0].getGreen(), _colors[0].getBlue());
020            }
021        }
022    }
