001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.AnalogChannel;
004    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
005    import edu.wpi.first.wpilibj.tables.ITable;
006    
007    public class StringPot implements LiveWindowSendable {
008        public final static double VAL_MAX_SAFE   = 6.00, // TODO: calibrate this
009                                   VAL_NEAR       = 3.00,
010                                   VAL_CENTER     = 2.67,
011                                   VAL_OPPAUTO    = 2.25,
012                                   VAL_FEEDER     = 1.81;
013        
014        private AnalogChannel _channel;
015        
016        public StringPot(int channelNum) {
017            _channel = new AnalogChannel(channelNum);
018        }
019        
020        public double get() {
021            double val = _channel.getAverageVoltage();
022            return val > 0.05 ? val : 0;
023        }
024        
025        private ITable _table = null;
026    
027        public void updateTable() {
028            if(_table != null) {
029                _table.putNumber("Value", get());
030            }
031        }
032        public void startLiveWindowMode() {}
033        public void stopLiveWindowMode() {}
034    
035        public void initTable(ITable subtable) {
036            _table = subtable;
037            updateTable();
038        }
039    
040        public ITable getTable() {
041            return _table;
042        }
043    
044        public String getSmartDashboardType() {
045            return "Analog Input";
046        }
047    }
