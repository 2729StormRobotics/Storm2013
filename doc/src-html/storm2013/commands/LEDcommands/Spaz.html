005    package storm2013.commands.LEDcommands;
006    
007    import edu.wpi.first.wpilibj.Timer;
008    import edu.wpi.first.wpilibj.command.Command;
009    import java.util.Random;
010    import storm2013.Robot;
011    import storm2013.subsystems.LEDStrip;
012    
013    /**
014     * Sets the {@link LEDStrip} color to some random color 20 times a second, or at 
015     * some other specified rate.
016     * @author evan1026
017     */
018    public class Spaz extends Command {
019        private static final double DEFAULT_PERIOD = 1.0/20;
020        private static final Color[] colors = {new Color(255,0,0), new Color(255,255,0), new Color(0,255,0),
021                                               new Color(0,255,255), new Color(0,0,255), new Color(255,0,255)};
022        
023        private double _period;
024        private Timer  _timer = new Timer();
025        private Random _rand  = new Random();
026        
027        public Spaz(){
028            this(DEFAULT_PERIOD);
029        }
030        
031        public Spaz(double period){
032            _period = period;
033            requires(Robot.ledStrip);
034        }
035        
036        protected void initialize() {
037            _timer.start();
038        }
039    
040        protected void execute() {
041            double time = _timer.get();
042            
043            if (time >= _period){
044                Color c = colors[_rand.nextInt(colors.length)];
045                Robot.ledStrip.setColor(c.getRed(),c.getBlue(),c.getGreen());
046                _timer.reset();
047            }
048        }
049    
050        protected boolean isFinished() {
051            return false;
052        }
053    
054        protected void end() {
055        }
056    
057        protected void interrupted() {
058            end();
059        }
060        
061    }
