001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.OI;
005    import storm2013.Robot;
006    import storm2013.subsystems.DriveTrain;
007    
008    public class ArcadeDrive extends Command {
009        DriveTrain driveTrain = Robot.driveTrain;
010        OI oi = Robot.oi;
011        
012        public ArcadeDrive() {
013            requires(Robot.driveTrain);
014        }
015    
016        protected void initialize() {}
017    
018        protected void execute() {
019            driveTrain.arcadeDrive(oi.getDriveAxis(), oi.getTurnAxis());
020        }
021    
022        protected boolean isFinished() {
023            return false;
024        }
025    
026        protected void end() {}
027    
028        protected void interrupted() {}
029    }
