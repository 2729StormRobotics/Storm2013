001    package storm2013.commands.LEDcommands;
002    
003    /**
004     * Color enumeration for the LED strip. Not exactly RGB, because of LED
005     * weirdness.
006     */
007    public class Color {
008        public static final Color OFF         = new Color(0,     0,   0);   //off
009        public static final Color MOVING      = new Color(255, 255, 255); //white
010        public static final Color SPINNING_UP = new Color(0,   255,   0);   //green
011        public static final Color FIRING      = new Color(255,   0,   0);   //red
012        
013        private int _red,
014                    _green,
015                    _blue;
016        
017        public Color(int red, int green, int blue){
018            _red = red;
019            _green = green;
020            _blue = blue;
021        }
022        
023        public int getRed(){
024            return _red;
025        }
026        public int getGreen(){
027            return _green;
028        }
029        public int getBlue(){
030            return _blue;
031        }
032    }
