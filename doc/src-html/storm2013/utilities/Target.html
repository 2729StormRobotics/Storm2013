001    package storm2013.utilities;
002    
003    /** Enumeration of different targets and axes for target tracking. */
004    public class Target{
005        public final static Target TwoPT = new Target("2pt Target ");
006        public final static Target ThreePT = new Target("3pt Target ");
007        public final static Target FivePT = new Target("5pt Target ");
008        
009        /** Different types of angles (X = horizontal,Y = vertical). */
010        public static class Axis {
011            public final static Axis X = new Axis("X Angle");
012            public final static Axis Y = new Axis("Y Angle");
013            
014            private String _value;
015            private Axis(String value) {
016                _value = value;
017            }
018            
019            public String getValue() {
020                return _value;
021            }
022        }
023    
024        private String _key;
025        private Target(String key){
026            _key = key;
027        }
028        
029        /** Returns the SmartDashboard key for a given Axis. */
030        public String getAxisKey(Axis axis) {
031            return _key + axis.getValue();
032        }
033        
034        /** Returns the SmartDashboard key for the X axis. */
035        public String getXKey(){
036            return getAxisKey(Axis.X);
037        }
038        
039        /** Returns the SmartDashboard key for the Y axis. */
040        public String getYKey(){
041            return getAxisKey(Axis.Y);
042        }
043        
044        /** Returns the SmartDashboard key for whether a target has been found. */
045        public String getFoundKey(){
046            return _key + "Found?";
047        }
048    }
