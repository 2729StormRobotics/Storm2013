001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.SpeedController;
004    import edu.wpi.first.wpilibj.buttons.Trigger;
005    import java.util.Timer;
006    import java.util.TimerTask;
007    
008    /**
009     * Uses two triggers (of any type) to keep a motor from spinning outside of a certain range.
010     * @author ginto
011     */
012    public class LimitSwitchedMotor implements SpeedController {
013        private static final double DEFAULT_PERIOD = 1.0/50;
014        
015        private final SpeedController _controller;
016        private final Trigger _top,_bottom;
017        private final boolean      _topOn,_bottomOn;
018        private double _value;
019        private Timer _bgThread = new Timer();
020        
021        private TimerTask _bgTask = new TimerTask() {
022            public void run() {
023                double output = _value;
024                if(output > 0 && _top != null) {
025                    if(_top.get() == _topOn) {
026                        output = 0;
027                    }
028                } else if(output < 0 && _bottom != null) {
029                    if(_bottom.get() == _bottomOn) {
030                        output = 0;
031                    }
032                }
033                _controller.set(output);
034            }
035        };
036    
037        // "top" being positive, "bottom" negative
038        /**
039         * 
040         * @param controller The {@link SpeedController} that controls the motor.
041         * @param top The top {@link Trigger} (trigger at the positive end of the motor's output).
042         * @param topOnValue The value from the top {@link Trigger} where you want the motor to stop.
043         * @param bottom The bottom {@link Trigger} (trigger at the negative end of the motor's output).
044         * @param bottomOnValue The value from the bottom {@link Trigger} where you want the motor to stop.
045         */
046        public LimitSwitchedMotor(SpeedController controller,Trigger top,boolean topOnValue,
047                                                             Trigger bottom,boolean bottomOnValue) {
048            _controller = controller;
049            _top = top;
050            _bottom = bottom;
051            _bgThread.schedule(_bgTask, 0, (long)(DEFAULT_PERIOD*1000));
052            _topOn = topOnValue;
053            _bottomOn = bottomOnValue;
054        }
055    
056        /**
057         * Gets the speed at which the motor is running.
058         * @return The current output speed
059         */
060        public double get() {
061            return _value;
062        }
063    
064        /**
065         * Sets a new output speed.
066         * @param speed The new speed
067         * @param syncGroup Honestly, I have no idea, but it's not used anyway
068         */
069        public void set(double speed, byte syncGroup) {
070            set(speed);
071        }
072    
073        /**
074         * Sets the new output speed
075         * @param speed The new speed
076         */
077        public void set(double speed) {
078            _value = speed;
079        }
080    
081        /**
082         * Stops the motor and disables its controller
083         */
084        public void disable() {
085            set(0);
086            _controller.disable();
087        }
088    
089        /**
090         * Writes values from the PID controller to the motor.
091         * @param output 
092         */
093        public void pidWrite(double output) {
094            set(output);
095        }
096        
097    }
