001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import edu.wpi.first.wpilibj.command.CommandGroup;
005    
006    /**
007     * Spins up the shooter, then spins the tomahawk once when it is up to speed. If
008     * it is interrupted partway through spinning the tomahawk, it will spawn
009     * another {@link SpinTomahawk} to finish the rotation.
010     */
011    public class Shoot extends CommandGroup {
012        private boolean _spinningTomahawk = false;
013        
014        private class _BeginTomahawk extends Command {
015            protected void initialize() {
016                _spinningTomahawk = true;
017            }
018            protected void execute() {}
019            protected boolean isFinished() {
020                return true;
021            }
022            protected void end() {}
023            protected void interrupted() {}
024        };
025        private class _EndTomahawk extends Command {
026            protected void initialize() {
027                _spinningTomahawk = false;
028            }
029            protected void execute() {}
030            protected boolean isFinished() {
031                return true;
032            }
033            protected void end() {}
034            protected void interrupted() {}
035        };
036        
037        public Shoot(double speed) {
038            addSequential(new _EndTomahawk());
039            addSequential(new SpinUp(speed));
040            addSequential(new _BeginTomahawk());
041            addSequential(new SpinTomahawk(true));
042            addSequential(new _EndTomahawk());
043        }
044        
045        public Shoot() {
046            this(SpinUp.SPEED_NORMAL);
047        }
048    
049        protected void interrupted() {
050            if(_spinningTomahawk) {
051                new SpinTomahawk(true).start();
052            }
053        }
054    }
