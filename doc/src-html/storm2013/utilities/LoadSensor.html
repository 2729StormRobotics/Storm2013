001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.AnalogChannel;
004    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
005    import edu.wpi.first.wpilibj.tables.ITable;
006    
007    /** Provides the ability to easily interface with the load sensor. */
008    public class LoadSensor implements LiveWindowSendable {
009        // slope: x-axis: amps y-axis: volts (Dropbox/Storm2729/current-sensor-curve.xlsx)
010        private final double voltToAmpRatio = 39.46e-3;
011        
012        private AnalogChannel loadSensor;
013        private ITable _table;
014        
015        public LoadSensor(int channel){
016            loadSensor = new AnalogChannel(channel);
017        }
018        
019        /** Returns (approximate) amperage being drawn. */
020        public double getAmps(){
021            return loadSensor.getVoltage() / voltToAmpRatio;
022        }
023        
024        /** Returns voltage drop being read. */
025        public double getVolts(){
026            return loadSensor.getVoltage();
027        }
028        
029        /** Puts current data into the {@link ITable}. */
030        public void updateTable() {
031            if(_table != null) {
032                _table.putNumber("voltage", loadSensor.getVoltage());
033                _table.putNumber("amperage", getAmps());
034            }
035        }
036    
037        public void startLiveWindowMode() {}
038        public void stopLiveWindowMode() {}
039    
040        /** Sets up an {@link ITable} for writing. */
041        public void initTable(ITable table) {
042            _table = table;
043            updateTable();
044        }
045    
046        /** Returns the currently used {@link ITable}. */
047        public ITable getTable() {
048            return _table;
049        }
050    
051        /** We have a custom widget, so it gets a custom type. */
052        public String getSmartDashboardType() {
053            return "Load Sensor";
054        }
055    }
