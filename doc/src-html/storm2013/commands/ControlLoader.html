001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.Victor;
004    import edu.wpi.first.wpilibj.command.Command;
005    import storm2013.RobotMap;
006    
007    /**
008     * Controls the loading thing in the back that pulls the balls up to the shooter.
009     * @author evan1026
010     */
011    public class ControlLoader extends Command {
012    
013        private Victor ballLiftMotor;
014        private double speed;
015        
016        /**
017         * Controls the loading thing in the back that pulls the balls up to the shooter.
018         * @param goingUp True to move the balls up or false to move them down.
019         */
020        public ControlLoader(boolean goingUp){
021            ballLiftMotor = new Victor(RobotMap.PORT_MOTOR_BALL_LIFT);
022            speed = (goingUp) ? -1 : 1;
023        }
024        
025        /**
026         * {@inheritDoc}
027         */
028        protected void initialize() {
029            ballLiftMotor.set(speed);
030        }
031    
032        /**
033         * {@inheritDoc}
034         */
035        protected void execute() {
036        }
037    
038        /**
039         * {@inheritDoc}
040         * @return {@inheritDoc}
041         */
042        protected boolean isFinished() {
043            return false;
044        }
045    
046        /**
047         * {@inheritDoc}
048         */
049        protected void end() {
050            ballLiftMotor.set(0);
051        }
052    
053        /**
054         * {@inheritDoc}
055         */
056        protected void interrupted() {
057            end();
058        }
059    }
