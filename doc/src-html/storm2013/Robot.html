001    package storm2013;
002    
003    import storm2013.commands.autonomous.JustShoot;
004    import edu.wpi.first.wpilibj.IterativeRobot;
005    import edu.wpi.first.wpilibj.command.Command;
006    import edu.wpi.first.wpilibj.command.Scheduler;
007    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
008    import edu.wpi.first.wpilibj.smartdashboard.SendableChooser;
009    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
010    import storm2013.commands.LEDcommands.SetModeRainbowDanceParty;
011    import storm2013.commands.SpinDown;
012    import storm2013.commands.autonomous.PyramidShoot;
013    import storm2013.subsystems.DriveTrain;
014    import storm2013.subsystems.LEDStrip;
015    import storm2013.subsystems.Shooter;
016    import storm2013.subsystems.Tilter;
017    import storm2013.subsystems.Tomahawk;
018    import storm2013.subsystems.Vision;
019    
020    /** 
021     * This is the robot's "Main class" which is run by the VM.
022     */
023    public class Robot extends IterativeRobot {
024        // All subsystems are accessible by Robot.name
025        public static OI         oi;
026        public static DriveTrain driveTrain;
027        public static Shooter    shooter;
028        public static Tilter     tilter;
029        public static Tomahawk   tomahawk;
030        public static LEDStrip   ledStrip;
031        public static Vision     vision;
032        
033        Command teleop;
034        String[] autonomiceNames;
035        Command[] autonomice;
036        SendableChooser chooser = new SendableChooser();
037        Command autonomouse;
038        
039        /** Called on robot boot. */
040        public void robotInit() {
041            driveTrain = new DriveTrain();
042            shooter    = new Shooter();
043            tilter     = new Tilter();
044            tomahawk   = new Tomahawk();
045            ledStrip   = new LEDStrip();
046            vision     = new Vision();
047            // Initialize OI last so it doesn't try to access null subsystems
048            oi         = new OI();
049    
050            // The names, and corresponding Commands of our autonomous modes
051            autonomiceNames = new String[]{"Rainbow Dance Party!", "Just shoot", "Shoot through Pyramid"};
052            autonomice = new Command[]{new SetModeRainbowDanceParty(), new JustShoot(), new PyramidShoot()};
053    
054            // Configure and send the SendableChooser, which allows autonomous modes
055            // to be chosen via radio button on the SmartDashboard
056            System.out.println(autonomice.length);
057            for (int i = 0; i < autonomice.length; ++i) {
058                chooser.addObject(autonomiceNames[i], autonomice[i]);
059            }
060            SmartDashboard.putData("Which Autonomouse?", chooser);
061    
062            // Put the Shooter PID on the dashboard so it can be enabled and stuff
063            SmartDashboard.putData("Shooter PID",shooter.getPIDController());
064            
065            // To make sure the Accelerator is running
066            shooter.enable();
067            // To make sure it isn't accelerating
068            new SpinDown().start();
069            
070            // Send sensor info to the SmartDashboard periodically
071            new Command("Sensor feedback") {
072                protected void initialize() {}
073                protected void execute() {
074                    SmartDashboard.putNumber("Wheel Speed (RPM)", shooter.getSpeedRpm());
075                    SmartDashboard.putNumber("Wheel signal", shooter.getMotorValRaw());
076                    SmartDashboard.putBoolean("Tomahawk forward?", tomahawk.isForward());
077                    SmartDashboard.putBoolean("Tilter at top?", tilter.isTopLimitTriggered());
078                    SmartDashboard.putNumber("String-pot", tilter.readStringPot());
079                }
080                protected boolean isFinished() {
081                    return false;
082                }
083                protected void end() {}
084                protected void interrupted() {
085                    end();
086                }
087            }.start();
088        }
089    
090        /** Called at the start of autonomous mode. */
091        public void autonomousInit() {
092            shooter.enable();
093            new SpinDown().start();
094            SmartDashboard.putBoolean("Enabled", true);
095            if (teleop != null) {
096                teleop.cancel();
097            }
098            autonomouse = (Command) chooser.getSelected();
099            if (autonomouse != null) {
100                autonomouse.start();
101            }
102        }
103    
104        /**
105         * Called during autonomous whenever a new driver station packet arrives
106         * (about every 1/50 of a second).
107         */
108        public void autonomousPeriodic() {
109            // Runs commands & stuff.
110            Scheduler.getInstance().run();
111        }
112    
113        /** Called at the start of teleop mode. */
114        public void teleopInit() {
115            shooter.enable();
116            new SpinDown().start();
117            SmartDashboard.putBoolean("Enabled", true);
118            if (autonomouse != null) {
119                autonomouse.cancel();
120            }
121            if (teleop != null) {
122                teleop.start();
123            }
124        }
125    
126        /**
127         * Called during teleop whenever a new driver station packet arrives (about
128         * every 1/50 of a second).
129         */
130        public void teleopPeriodic() {
131            // Runs commands & stuff
132            Scheduler.getInstance().run();
133        }
134    
135        /** Called at the start of test mode */
136        public void testInit() {
137            SmartDashboard.putBoolean("Enabled", false);
138        }
139    
140        /**
141         * Called during test whenever a new driver station packet arrives (about
142         * every 1/50 of a second).
143         */
144        public void testPeriodic() {
145            // Updates sensors & actuators on the LiveWindow
146            LiveWindow.run();
147        }
148    
149        /** Called after any of the other modes ends. */
150        public void disabledInit() {
151            SmartDashboard.putBoolean("Enabled", false);
152            if(autonomouse != null) {
153                autonomouse.cancel();
154            }
155            if(teleop != null) {
156                teleop.cancel();
157            }
158            // Make sure the shooter doesn't start spinning again when the robot
159            // re-enables.
160            shooter.setSetpoint(0);
161            shooter.setMotorValRaw(0);
162        }
163        /**
164         * Called during disabled whenever a new driver station packet arrives
165         * (about every 1/50 of a second). We only have it overridden so we don't
166         * get "Override me!" messages.
167         */
168        public void disabledPeriodic() {}
169    }
