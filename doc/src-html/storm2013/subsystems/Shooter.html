005    package storm2013.subsystems;
006    
007    import edu.wpi.first.wpilibj.DigitalInput;
008    import edu.wpi.first.wpilibj.Counter;
009    import edu.wpi.first.wpilibj.CounterBase;
010    
011    import storm2013.utilities.IEncoderSpeed;
012    
013    import storm2013.RobotMap;
014    
015    /**
016     *
017     * @author Storm
018     */
019    public class Shooter implements IEncoderSpeed{
020        
021        DigitalInput hallEffect;
022        Counter counter;
023        
024        /**
025         * @see IEncoder
026         */
027        public Shooter(){
028            
029            hallEffect = new DigitalInput(RobotMap.PORT_HALL_EFFECT);
030            counter = new Counter(  CounterBase.EncodingType.k1X, //Count only rising edge of digital signal
031                                    hallEffect,
032                                    hallEffect,
033                                    false); //inverted
034            
035            //counter.clearDownSource();
036            //counter.setUpSourceEdge(true, false); //TODO Check without this
037            counter.start();
038        }
039        
040        
041        /**
042         * 
043         * @return revolutions per minute
044         */
045        public double getRPM() {
046            return 60 /counter.getPeriod();
047        }
048        
049    }
