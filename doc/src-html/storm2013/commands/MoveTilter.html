001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.OI;
005    import storm2013.Robot;
006    
007    /**
008     * Moves the tilter based on user input as configured in {@link OI}
009     * @author Joe
010     */
011    public class MoveTilter extends Command {
012        
013        /**
014         * The constructor.
015         */
016        public MoveTilter() {
017            requires(Robot.tilter);
018        }
019    
020        /**
021         * Initializes command. Doesn't actually do anything.
022         */
023        protected void initialize() {}
024    
025        /**
026         * Executed periodically. It reads input values and applies them to the tilter.
027         */
028        protected void execute() {
029            Robot.tilter.move(Robot.oi.getTilterAxis()*Robot.tilter.SPEED_DEFAULT);
030        }
031    
032        /**
033         * This command is to be used continuously so this just returns false, indicating
034         * that it is never finished.
035         * @return false
036         */
037        protected boolean isFinished() {
038            return false;
039        }
040    
041        /**
042         * Called when the command is ending. Doesn't do anything in this case.
043         */
044        protected void end() {}
045    
046        /**
047         * Called when the command is interrupted. Calls {@link #end()}
048         */
049        protected void interrupted() {
050            end();
051        }
052    }
