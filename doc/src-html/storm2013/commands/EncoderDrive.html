005    package storm2013.commands;
006    
007    import edu.wpi.first.wpilibj.command.Command;
008    import storm2013.Robot;
009    import storm2013.subsystems.DriveTrain;
010    
011    /**
012     * Drives a certain number of wheel rotations based on an encoder. The encoder values
013     * are supplied by the {@link IDriveTrainEncoder}, which is implemented by some other class
014     * to supply values in rotations. It then drives at the specified drive speed until
015     * until it reaches the goal.
016     * 
017     * @author evan1026
018     */
019    
020    public class EncoderDrive extends Command {
021    
022        private double _goal;
023        private double _dist;
024        private double _driveSpeed;
025        private double _decelThreshold;
026        private double _stopThreshold;
027        private IDriveTrainEncoder _encoder;
028        private DriveTrain _driveTrain = Robot.driveTrain;
029        
030        /**
031         * Main constructor for the class. More could be created if it proves to be worthwhile,
032         * but for now, use this one.
033         * 
034         * @param goal            How far you want to go (use rotations for units, just to make a standard).
035         * @param driveSpeed      The speed at which you want to travel (also in rotations).
036         * @param encoder         The class that implements {@link IDriveTrainEncoder}.
037         * @param decelThreshold  The threshold to determine when to start to decelerate.
038         * @param stopThreshold   The threshold to determine when to stop the command.
039         * @see   IDriveTrainEncoder
040         */
041        public EncoderDrive(double goal, double driveSpeed, IDriveTrainEncoder encoder,
042                double decelThreshold, double stopThreshold){ 
043            
044            if (encoder == null){
045                throw new NullPointerException("The DriveTrainEncoder cannot be null.");
046            }
047            
048            _goal = goal;
049            _driveSpeed = driveSpeed;
050            _encoder = encoder;
051            _decelThreshold = decelThreshold;
052            _stopThreshold = stopThreshold;
053            requires(Robot.driveTrain);
054        }
055        
056        /**
057         * Implemented from {@link Command}, but it doesn't actually do anything at the moment.
058         */
059         protected void initialize() {
060        }
061        
062        /**
063         * Implemented from {@link Command}. It is run over and over and such, and it calculates
064         * what the drive speed the robot needs to accelerate toward by setting values in 
065         * the {@link DriveTrain}.
066         * 
067         * @see DriveTrain
068         */
069        public void execute() {
070            _dist = _encoder.getDistance();
071            
072            if(Math.abs(_goal) - Math.abs(_dist) > _decelThreshold){
073                if (_dist < _goal){
074                    _driveTrain.tankDrive(_driveSpeed, _driveSpeed);
075                }
076                else if (_dist > _goal){
077                    _driveTrain.tankDrive(-_driveSpeed, -_driveSpeed);
078                }
079            }
080            else{
081                _driveTrain.tankDrive(0, 0);
082            }  
083        }
084        
085        /**
086         * Implemented from {@link Command}. When it returns true, it runs {@link #end()},
087         * and then stops running the command.
088         * 
089         * @return true if it is within the stopping threshold ({@link #_stopThreshold}) and false otherwise.
090         */
091        public boolean isFinished() {
092            return (Math.abs(_goal) - Math.abs(_dist) <= _stopThreshold);
093        }
094        
095        /**
096         * Implemented from {@link Command}. This is the last thing run before the command
097         * stops running. At the moment, it doesn't actually do anything.
098         */
099        public void end() {
100        }
101        
102        /**
103         * Implemented from {@link Command}. When {@link Command#cancel()} is run, 
104         * this is first called, then {@link #end()}.
105         */
106        public void interrupted() {
107        }
108    
109        /**
110         * Gets the current goal.
111         * @return The current goal, in rotations.
112         */
113        public double getGoal() {
114            return _goal;
115        }
116        
117        /**
118         * Sets a new goal.
119         * @param goal The new goal, in rotations.
120         */
121        public void setGoal(double goal) {
122            _goal = goal;
123        }
124        
125        /**
126         * Gets the total distance traveled.
127         * @return The distance traveled, in revolutions.
128         */
129        public double getDist() {
130            return _dist;
131        }
132       
133        //No set distance. You should implement that in the encoder.
134       
135        /**
136         * Returns the speed at which the robot it set to drive.
137         * @return The drive speed, in RPM.
138         */
139        public double getDriveSpeed() {
140            return _driveSpeed;
141        }
142        
143        /**
144         * Sets the drive speed.
145         * @param driveSpeed The drive speed, in RPM.
146         */
147        public void setDriveSpeed(double driveSpeed) {
148            _driveSpeed = driveSpeed;
149        }
150        
151        /**
152         * Gets the threshold at which the robot will start to decelerate.
153         * @return The threshold, a distance, in revolutions.
154         */
155        public double getDecelThreshold() {
156            return _decelThreshold;
157        }
158        
159        /**
160         * Sets a new threshold to decide when the robot will decelerate.
161         * @param decelThreshold The new deceleration threshold, in revolutions.
162         */
163        public void setDecelThreshold(double decelThreshold) {
164            _decelThreshold = decelThreshold;
165        }
166        
167        /**
168         * Gets the threshold at which the robot will stop executing the command.
169         * @return The threshold, in revolutions.
170         */
171        public double getStopThreshold() {
172            return _stopThreshold;
173        }
174        
175        /**
176         * Sets a new threshold at which the robot will stop.
177         * @param stopThreshold The new threshold, in revolutions
178         */
179        public void setStopThreshold(double stopThreshold) {
180            _stopThreshold = stopThreshold;
181        }
182        
183        
184    }
