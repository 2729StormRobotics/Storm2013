001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.CommandGroup;
004    
005    /**
006     * Allows a user to control the feeder given a direction to spin and a time.
007     * @author evan1026
008     */
009    public class ControlFeeder extends CommandGroup {
010        /**
011         * Allows a user to control the feeder given a direction to spin and a time.
012         * @param goingUp True to load balls in (move them up) or false to push them out (move them down).
013         * @param time    The amount of time (in seconds) for the feeder to spin.
014         */
015        public ControlFeeder(boolean goingUp, int time){
016            addSequential(new StartFeeder(goingUp));
017            addSequential(new DoNothing(), time);
018            addSequential(new StopFeeder());
019        }
020    }
