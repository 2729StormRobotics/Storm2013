001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.Counter;
004    import edu.wpi.first.wpilibj.CounterBase;
005    import edu.wpi.first.wpilibj.DigitalInput;
006    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
007    import edu.wpi.first.wpilibj.tables.ITable;
008    
009    /**
010     * This uses a digital signal from a hall effect sensor to calculate the speed
011     * of a wheel.
012     * @author Joe
013     */
014    public class HallEffectSpeedSensor implements LiveWindowSendable,ISpeedSensor {
015        private DigitalInput _hallEffect;
016        private Counter _counter;
017    
018        /**
019         * 
020         * @param port The port for the hall effect sensor's digital input
021         */
022        public HallEffectSpeedSensor(int port) {
023            _hallEffect = new DigitalInput(port);
024            // TODO: Try this as a replacement for the rest of the counter initialization
025            _counter = new Counter(_hallEffect);
026    //        _counter = new Counter(CounterBase.EncodingType.k1X, //Count only rising edge of digital signal
027    //                               _hallEffect,
028    //                               _hallEffect,
029    //                               false); //inverted
030    //
031    //        _counter.clearDownSource();
032    //        _counter.setUpSourceEdge(true, false); //TODO Check without this
033            _counter.start();
034        }
035        
036        public void setMinSpeedRpm(double speedRpm) {
037            _counter.setMaxPeriod(60/speedRpm);
038        }
039        
040        public double getSpeedRpm() {
041            return 60/_counter.getPeriod();
042        }
043    
044        ITable _table;
045        public void initTable(ITable table) {
046            _table = table;
047            updateTable();
048        }
049    
050        public ITable getTable() {
051            return _table;
052        }
053    
054        public String getSmartDashboardType() {
055            return "Analog Input";
056        }
057    
058        public void updateTable() {
059            if(_table != null) {
060                _table.putNumber("Value", getSpeedRpm());
061            }
062        }
063    
064        public void startLiveWindowMode() {}
065        public void stopLiveWindowMode() {}
066    
067    }
