001    package storm2013.commands.LEDcommands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /** Sets LED color, then leaves it that way until interrupted somehow. */
007    public class SetColor extends Command {
008        private int _red,
009                    _green,
010                    _blue;
011        
012        public SetColor(int red, int green, int blue){
013            _red = red;
014            _green = green;
015            _blue = blue;
016            requires(Robot.ledStrip);
017        }
018        
019        public SetColor(Color color){
020            this(color.getRed(),color.getGreen(),color.getBlue());
021        }
022        
023        protected void initialize() {
024            Robot.ledStrip.setColor(_red, _green, _blue);
025        }
026        protected void execute() {}
027        protected boolean isFinished() {
028            return false;
029        }
030        protected void end() {}
031        protected void interrupted() {
032            end();
033        }
034        
035    }
