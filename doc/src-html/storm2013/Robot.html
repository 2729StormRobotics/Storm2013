001    package storm2013;

003    import edu.wpi.first.wpilibj.IterativeRobot;
004    import edu.wpi.first.wpilibj.command.Command;
005    import edu.wpi.first.wpilibj.command.Scheduler;
006    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
007    import edu.wpi.first.wpilibj.smartdashboard.SendableChooser;
008    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
009    import storm2013.commands.*;
010    import storm2013.subsystems.DriveTrain;
011    import storm2013.subsystems.Shooter;
012    import storm2013.subsystems.Tilter;
013    import storm2013.subsystems.Tomahawk;
014    import storm2013.utilities.LoadSensor;
015    
016    /**
017     * The VM is configured to automatically run this class, and to call the
018     * functions corresponding to each mode, as described in the IterativeRobot
019     * documentation. If you change the name of this class or the package after
020     * creating this project, you must also update the manifest file in the resource
021     * directory.
022     */
023    public class Robot extends IterativeRobot {
024        public static OI oi;
025        public static DriveTrain driveTrain;
026        public static Shooter shooter;
027        public static Tilter tilter;
028        public static Tomahawk tomahawk;
029        
030        Command teleop;
031        String[] autonomiceNames;
032        Command[] autonomice;
033        SendableChooser chooser = new SendableChooser();
034        Command autonomouse;
035        
036        LoadSensor loadSensor;
037        
038    //    DigitalInput irTest = new DigitalInput(7);
039    
040        public void robotInit() {
041            driveTrain = new DriveTrain();
042            shooter    = new Shooter();
043            tilter     = new Tilter();
044            tomahawk   = new Tomahawk();
045            oi         = new OI();
046    
047            loadSensor = new LoadSensor(2);
048            
049            autonomiceNames = new String[]{"Do Nothing", "DriveStuff"};
050    
051            autonomice = new Command[]{new DoNothing(), new DriveStuff()};
052    
053            System.out.println(autonomice.length);
054            for (int i = 0; i < autonomice.length; ++i) {
055                chooser.addObject(autonomiceNames[i], autonomice[i]);
056            }
057            SmartDashboard.putData("Which Autonomouse?", chooser);
058    
059    //        SmartDashboard.putData(Scheduler.getInstance());
060    
061            LiveWindow.addSensor("Load Sensor", "Load Sensor 1", loadSensor);
062    //        LiveWindow.addSensor("Tilter","DIO 7", irTest);
063        }
064    
065        public void autonomousInit() {
066            if (teleop != null) {
067                teleop.cancel();
068            }
069            autonomouse = (Command) chooser.getSelected();
070            if (autonomouse != null) {
071                autonomouse.start();
072            }
073        }
074    
075        public void autonomousPeriodic() {
076            SmartDashboard.putData("Load Sensor", loadSensor);
077            Scheduler.getInstance().run();
078        }
079    
080        public void teleopInit() {
081            if (autonomouse != null) {
082                autonomouse.cancel();
083            }
084            if (teleop != null) {
085                teleop.start();
086            }
087        }
088    
089        public void teleopPeriodic() {
090            Scheduler.getInstance().run();
091            SmartDashboard.putNumber("Tilter angle", tilter.getAngle());
092            SmartDashboard.putNumber("Wheel Speed (RPM)", shooter.getSpeedRpm());
093            SmartDashboard.putNumber("Gyro angle", driveTrain.getGyroAngle());
094        }
095    
096        public void testPeriodic() {
097            LiveWindow.run();
098        }
099    
100        public void disabledInit() {
101            if(autonomouse != null) {
102                autonomouse.cancel();
103            }
104            if(teleop != null) {
105                teleop.cancel();
106            }
107        }
108    
109        // Eliminates "Overload me!" messages
110        public void disabledPeriodic() {
111        }
112    }
