001    package storm2013.subsystems;
002    
003    import edu.wpi.first.wpilibj.DigitalInput;
004    import edu.wpi.first.wpilibj.Victor;
005    import edu.wpi.first.wpilibj.command.Subsystem;
006    import storm2013.RobotMap;
007    
008    /** Subsystem including the Tomahawk and its IR sensor. */
009    public class Tomahawk extends Subsystem {
010        private static double MOTOR_SPEED = 1;
011        
012        private Victor _motor;
013        private DigitalInput _limitSwitch;
014        
015        public Tomahawk() {
016            _motor = new Victor(RobotMap.PORT_MOTOR_TOMAHAWK);
017            _limitSwitch = new DigitalInput(RobotMap.PORT_LIMIT_TOMAHAWK);
018        }
019    
020        protected void initDefaultCommand() {}
021        
022        /** Reads whether the tomahawk is forward (not triggering the IR sensor). */
023        public boolean isForward() {
024            return _limitSwitch.get();
025        }
026        
027        /** Moves the tomahawk (true = normal direction,false = reverse). */
028        public void move(boolean forward) {
029            _motor.set((forward ? 1 : -1)*MOTOR_SPEED);
030        }
031        
032        /** Stops the tomahawk. */
033        public void stop() {
034            _motor.set(0);
035        }
036    }
