001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /**
007     * Spins down the firing wheel.
008     * @author evan1026
009     */
010    public class SpinDown extends Command {
011    
012        /**
013         * Creates a new instance of the command.
014         */
015        public SpinDown(){
016            requires(Robot.shooter);
017        }
018        
019        /**
020         * Initializes the {@link Command}
021         */
022        protected void initialize() {
023            Robot.shooter.getPIDController().setSetpoint(0);
024        }
025    
026        /**
027         * Does nothing
028         */
029        protected void execute() {
030        }
031    
032        /**
033         * Tells if the {@link Command} is done
034         * @return If the shooter is at its target.
035         */
036        protected boolean isFinished() {
037            return Robot.shooter.onTarget();
038        }
039    
040        /**
041         * Ends the {@link Command}
042         */
043        protected void end() {
044        }
045    
046        /**
047         * Also ends the {@link Command}
048         */
049        protected void interrupted() {
050            end();
051        }
052    }
