001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    import storm2013.subsystems.DriveTrain;
006    
007    /** Turns the robot until a desired heading is reached. */
008    public class GyroTurn extends Command {
009    
010        private double _goal;
011        private double _dist;
012        private double _turnSpeed;
013        
014        private DriveTrain _driveTrain = Robot.driveTrain;
015        
016        /**
017         * @param goal      Desired heading
018         * @param turnSpeed Speed at which to turn
019         */
020        public GyroTurn(double goal, double turnSpeed){
021            _goal = goal;
022            _turnSpeed = turnSpeed;
023            requires(Robot.driveTrain);
024        }
025        protected void initialize() {
026            Robot.driveTrain.clearGyro();
027        }
028        protected void execute() {
029            _dist = Robot.driveTrain.getGyroAngle();
030            
031            if (_goal < 0){
032                _driveTrain.tankDrive(-_turnSpeed, _turnSpeed);
033            }
034            else{
035                _driveTrain.tankDrive(_turnSpeed, -_turnSpeed);
036            }
037        }
038        protected boolean isFinished() {
039            if(_goal < 0) {
040                return _dist <= _goal;
041            }
042            else {
043                return _dist >= _goal;
044            }
045        }
046        protected void end() {
047            Robot.driveTrain.tankDrive(0, 0);
048        }
049        protected void interrupted() {
050            end();
051        }
052    }
