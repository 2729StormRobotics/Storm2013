009    /**
010     * This class is the glue that binds the controls on the physical operator
011     * interface to the commands and command groups that allow control of the robot.
012     */
013    public class OI {
014        //// CREATING BUTTONS
015        // One type of button is a joystick button which is any button on a joystick.
016        // You create one by telling it which joystick it's on and which button
017        // number it is.
018        // Joystick stick = new Joystick(port);
019        // Button button = new JoystickButton(stick, buttonNumber);
020        
021        // Another type of button you can create is a DigitalIOButton, which is
022        // a button or switch hooked up to the cypress module. These are useful if
023        // you want to build a customized operator interface.
024        // Button button = new DigitalIOButton(1);
025        
026        // There are a few additional built in buttons you can use. Additionally,
027        // by subclassing Button you can create custom triggers and bind those to
028        // commands the same as any other Button.
029        
030        //// TRIGGERING COMMANDS WITH BUTTONS
031        // Once you have a button, it's trivial to bind it to a button in one of
032        // three ways:
033        
034        // Start the command when the button is pressed and let it run the command
035        // until it is finished as determined by it's isFinished method.
036        // button.whenPressed(new ExampleCommand());
037        
038        // Run the command while the button is being held down and interrupt it once
039        // the button is released.
040        // button.whileHeld(new ExampleCommand());
041        
042        // Start the command when the button is released  and let it run the command
043        // until it is finished as determined by it's isFinished method.
044        // button.whenReleased(new ExampleCommand());
045        
046        public Joystick driveJoystick = new Joystick(RobotMap.PORT_JOYSTICK_DRIVE);
047        
048        private double _zeroDeadzone(double joyValue) {
049            return Math.abs(joyValue) > 0.1 ? joyValue : 0;
050        }
051        
052        private double _applyExponential(double joyValue) {
053            if(joyValue == 0) {
054                return 0;
055            }
056            double sign = (joyValue > 0) ? 1 : -1;
057            return sign*joyValue*joyValue/Math.sqrt(Math.abs(joyValue));
058        }
059        
060        public double getDriveAxis() {
061            return _applyExponential(_zeroDeadzone(-driveJoystick.getRawAxis(2)));
062        }
063        
064        public double getTurnAxis() {
065            return _zeroDeadzone(driveJoystick.getRawAxis(3));
066        }
067    }
