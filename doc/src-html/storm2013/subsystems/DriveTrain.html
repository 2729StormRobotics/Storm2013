001    package storm2013.subsystems;
002    
003    import edu.wpi.first.wpilibj.Encoder;
004    import edu.wpi.first.wpilibj.Gyro;
005    import edu.wpi.first.wpilibj.Jaguar;
006    import edu.wpi.first.wpilibj.RobotDrive;
007    import edu.wpi.first.wpilibj.command.Subsystem;
008    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
009    import storm2013.RobotMap;
010    import storm2013.commands.TankDrive;
011    
012    /** Subsystem including the drive motors, encoders, and gyro. */
013    
014    public class DriveTrain extends Subsystem {
015        
016        Jaguar _left  = new Jaguar(RobotMap.PORT_MOTOR_DRIVE_LEFT),
017               _right = new Jaguar(RobotMap.PORT_MOTOR_DRIVE_RIGHT);
018        
019        RobotDrive _drive = new RobotDrive(_left,_right);
020        
021        Encoder _leftEncoder =  new Encoder(RobotMap.PORT_ENCODER_LEFT_1, RobotMap.PORT_ENCODER_LEFT_2);
022        Encoder _rightEncoder = new Encoder(RobotMap.PORT_ENCODER_RIGHT_1, RobotMap.PORT_ENCODER_RIGHT_2);
023         
024        Gyro _gyro = new Gyro(RobotMap.PORT_SENSOR_GYRO);
025        
026        public DriveTrain() {
027            
028            _leftEncoder.start();
029            _rightEncoder.start();
030            
031            LiveWindow.addSensor  ("Drive Train", "Gyro", _gyro);
032            LiveWindow.addSensor  ("Drive Train", "Left Encoder", _leftEncoder);
033            LiveWindow.addSensor  ("Drive Train", "Right Encoder", _rightEncoder);
034        }
035        
036        protected void initDefaultCommand() {
037            setDefaultCommand(new TankDrive());
038        }
039        
040        /**
041         * Drives in arcade mode (think the old game "Asteroid").
042         * @param speed Forward/back (1 = forward,-1 = back)
043         * @param turn  (Counter)Clockwise (1 = clockwise,-1 = counterclockwise)
044         */
045        public void arcadeDrive(double speed,double turn) {
046            _drive.arcadeDrive(-speed, turn, false);
047        }
048        
049        /** Drives in tank mode (each motor gets 1 = forward,-1 = back). */
050        public void tankDrive(double leftValue, double rightValue){
051            _drive.tankDrive(-leftValue, -rightValue);
052        }
053    
054        /** Reads the left encoder (+ = forward,- = back). */
055         public double getLeftDistance() {
056            return _leftEncoder.get();
057        }
058    
059         /** Reads the right encoder (+ = forward,- = back). */
060        public double getRightDistance() {
061            return -_rightEncoder.get();
062        }
063        
064        /** Resets both encoders. */
065        public void clearEncoder(){
066            _leftEncoder.reset();
067            _rightEncoder.reset();
068        }
069        
070        /** Resets the gyro. */
071        public void clearGyro(){
072            _gyro.reset();
073        }
074        
075        /**
076         * Reads the Robot's (relative) heading from gyro (+ = clockwise,
077         * - = counterclockwise).
078         */
079        public double getGyroAngle(){
080            return _gyro.getAngle();
081        }
082        
083        /** Reads the [-1,1] value set for the left wheels. */
084        public double getLeftSpeed(){
085            return _left.get();
086        }
087        
088        /** Reads the [-1,1] value set for the right wheels. */
089        public double getRightSpeed(){
090            return _right.get();
091        }
092    }
