001    package storm2013.commands.autonomous;

003    import edu.wpi.first.wpilibj.command.Command;
004    import edu.wpi.first.wpilibj.command.CommandGroup;
005    import edu.wpi.first.wpilibj.command.StartCommand;
006    import storm2013.Robot;
007    import storm2013.commands.GyroTurn;
008    import storm2013.commands.LEDcommands.Color;

010    import storm2013.commands.LowerTilter;

013    import storm2013.commands.TargetPIDTilt;
014    import storm2013.commands.TiltSetDistance;
015    import storm2013.subsystems.Tilter;
016    import storm2013.subsystems.Vision;
017    import storm2013.utilities.Target;
018    
019    /**
020     * This autonomous mode moves back and shoots; we use it from the back-center
021     * of the pyramid.
022     */
023    public class PyramidShoot extends CommandGroup {
024        
025        public PyramidShoot() {
026            // Drive back. This used to be an EncoderDrive, but our right encoder
027            // is broken.
028            addSequential(new Command() {
029                {
030                    requires(Robot.driveTrain);
031                } 
032                protected void initialize() {
033                    Robot.driveTrain.clearEncoder();
034                }
035                protected void execute() {
036                    Robot.driveTrain.tankDrive(-0.6, -0.6);
037                }
038                protected boolean isFinished() {
039                    return Robot.driveTrain.getLeftDistance() < -1476;
040                }
041                protected void end() {
042                    Robot.driveTrain.tankDrive(0, 0);
043                }
044                protected void interrupted() {
045                    end();
046                }
047            });
048            addSequential(new StartCommand(new SetColor(new Color(0,0,255))));
049            addSequential(new TiltSetDistance(Tilter.SPEED_DEFAULT, Vision.Distance.NEAR));
050            // Align tilter with target
051            addSequential(new TargetPIDTilt(Target.ThreePT, 1.0, false));
052            // Shoot repreatedly (in case of jams)
053            addSequential(new Shoot());
054            addSequential(new Shoot());
055            addSequential(new Shoot());
056            addSequential(new Shoot());
057            addSequential(new Shoot());
058            addSequential(new Shoot());
059            // End shooting (not always necessary)
060            addSequential(new SpinDown());
061        }
062    }
