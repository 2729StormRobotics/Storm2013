001    package storm2013.subsystems;
002    
003    import edu.wpi.first.wpilibj.command.Subsystem;
004    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
005    import storm2013.commands.LEDcommands.SetModeMoving;
006    import storm2013.utilities.LED;
007    
008    /** Subsystem to control the LED strip. */
009    public class LEDStrip extends Subsystem {
010        LED _red,_green,_blue;
011    
012        public LEDStrip() {
013            _red   = new LED(6);
014            _green = new LED(8);
015            _blue  = new LED(7);
016            LiveWindow.addActuator("LED", "Red",   _red);
017            LiveWindow.addActuator("LED", "Green", _green);
018            LiveWindow.addActuator("LED", "Blue",  _blue);
019        }
020        
021        protected void initDefaultCommand() {
022            setDefaultCommand(new SetModeMoving());
023        }
024        
025        /**
026         * Sets the tri-color LEDs' color with red, green, and blue signals in the
027         * [0,255] range. This is NOT a true RGB color, because the lights respond
028         * weirdly to different signals.
029         */
030        public void setColor(int red, int green, int blue){
031            _red.set(red);
032            _green.set(green);
033            _blue.set(blue);
034        }
035        
036        /** Read current red value. */
037        public int getRed(){
038            return _red.get();
039        }
040        /** Read current green value. */
041        public int getGreen(){
042            return _green.get();
043        }
044        /** Read current blue value. */
045        public int getBlue(){
046            return _blue.get();
047        }
048        
049    }
