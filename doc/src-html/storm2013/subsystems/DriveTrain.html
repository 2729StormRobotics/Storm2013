001    package storm2013.subsystems;
002    
003    import edu.wpi.first.wpilibj.Encoder;
004    import edu.wpi.first.wpilibj.Gyro;
005    import edu.wpi.first.wpilibj.Jaguar;
006    import edu.wpi.first.wpilibj.RobotDrive;
007    import edu.wpi.first.wpilibj.command.Subsystem;
008    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
009    import storm2013.RobotMap;
010    import storm2013.commands.TankDrive;
011    
012    /**
013     * The subsystem that controls the drive train of the robot. It contains various
014     * methods that easily calculate the movement to make up for weird things in the robot,
015     * and it contains references to all of the devices that the robot has associated with
016     * the drive train.
017     * @author Joe
018     */
019    
020    public class DriveTrain extends Subsystem{
021        
022    //    private double ACCEL_RATE =1.7;
023    //    private double DECEL_RATE = 1.7;
024        
025        Jaguar _left  = new Jaguar(RobotMap.PORT_MOTOR_DRIVE_LEFT),
026               _right = new Jaguar(RobotMap.PORT_MOTOR_DRIVE_RIGHT);
027        
028    //    AccelerationLimiter _leftControl = new AccelerationLimiter(_left, ACCEL_RATE, DECEL_RATE);
029    //    AccelerationLimiter _rightControl = new AccelerationLimiter(_right, ACCEL_RATE, DECEL_RATE);
030        
031        RobotDrive _drive = new RobotDrive(_left,_right);
032        
033        Encoder _leftEncoder =  new Encoder(RobotMap.PORT_ENCODER_LEFT_1, RobotMap.PORT_ENCODER_LEFT_2);
034        Encoder _rightEncoder = new Encoder(RobotMap.PORT_ENCODER_RIGHT_1, RobotMap.PORT_ENCODER_RIGHT_2);
035         
036        Gyro _gyro = new Gyro(RobotMap.PORT_SENSOR_GYRO);
037        
038        /**
039         * The constructor which simply enables the encoders and adds the gyro to the {@link LiveWindow}
040         */
041        public DriveTrain() {
042            
043            _leftEncoder.start();
044            _rightEncoder.start();
045            
046    //        LiveWindow.addActuator("Drive Train", "Left Motor", _left);
047    //        LiveWindow.addActuator("Drive Train", "Right Motor", _right);
048            LiveWindow.addSensor  ("Drive Train", "Gyro", _gyro);
049        }
050        
051        /**
052         * Sets the default command for the drive train to run to be {@link TankDrive}.
053         */
054        public void initDefaultCommand() {
055            setDefaultCommand(new TankDrive());
056        }
057        
058        /**
059         * Moves the robot based on a speed value and a turn value, kind of like using
060         * a joystick with 2 axes on an arcade game.
061         * @param speed The speed at which to drive. On a joystick, this would come from the forward/backward axis
062         * @param turn The amount to turn. On a joystick, the left/right axis value.
063         */
064        public void arcadeDrive(double speed,double turn) {
065            _drive.arcadeDrive(-speed, turn, false);
066    //      _drive.tankDrive(-speed, -speed);
067        }
068        
069        /**
070         * Gives specific values to the left and right motors.
071         * @param leftValue The value to output to the left motor
072         * @param rightValue The value to output to the right motor
073         */
074        public void tankDrive(double leftValue, double rightValue){
075            _drive.tankDrive(-leftValue, -rightValue);
076        }
077    
078        /**
079         * Gets the reading on the left encoder
080         * @return The encoder value
081         */
082         public double getLeftDistance() {
083            return _leftEncoder.get();
084        }
085    
086         /**
087          * Gets the reading on the right encoder. Negated because the 2 encoders face opposite directions
088          * @return The encoder value
089          */
090        public double getRightDistance() {
091            return -_rightEncoder.get();
092        }
093        
094        /**
095         * Resets the values of both encoders.
096         */
097        public void clearEncoder(){
098            _leftEncoder.reset();
099            _rightEncoder.reset();
100        }
101        
102        /**
103         * Resets the gyro value.
104         */
105        public void clearGyro(){
106            _gyro.reset();
107        }
108        
109        /**
110         * Returns the gyro's angle. Negated due to the gyro being upside down
111         * @return The angle the robot has turned
112         */
113        public double getGyroAngle(){
114            return -_gyro.getAngle();
115        }
116    }
