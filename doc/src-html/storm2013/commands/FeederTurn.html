001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.CommandGroup;
004    import storm2013.Robot;
005    
006    /**
007     * Turn back/forth from feeder station. Interrupted if either analog stick is
008     * moved.
009     */
010    public class FeederTurn extends CommandGroup { 
011        public FeederTurn(boolean toward) {
012            addSequential(new GyroTurn((toward ? 1 : -1)*110, 0.65));
013        }
014        
015        protected boolean isFinished() {
016            return super.isFinished() || Robot.oi.getLeftDrive() != 0 || Robot.oi.getRightDrive() != 0;
017        }
018    }
