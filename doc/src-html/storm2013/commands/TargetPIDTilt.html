001    package storm2013.commands;
002    
003    import storm2013.Robot;
004    import storm2013.utilities.Target;
005    
006    /**
007     * Tilts the tilter based on the camera stuff
008     * @author evan1026
009     */
010    public class TargetPIDTilt extends TargetPIDCommand {
011        /**
012         * Constructor.
013         * @param target The {@link Target} to aim at
014         * @param timeout The timeout
015         * @param continuous Whether or not the command should be continuous
016         */
017        public TargetPIDTilt(Target target, double timeout, boolean continuous){
018            super(target,Target.Axis.Y,timeout,continuous,
019                  0, 0, 0, // P, I, D
020                  1,10);
021            requires(Robot.tilter);
022        }
023        
024        /**
025         * Returns the tilter angle
026         * @return the angle of the tilter
027         */
028        protected double returnPIDInput() {
029            return Robot.tilter.getAngle();
030        }
031    
032        /**
033         * Writes the output (divided by 10) to the tilter.
034         * @param output the output to write
035         */
036        protected void writePIDOut(double output) {
037            output /= 10;
038            Robot.tilter.move(output);
039        }
040    
041        /**
042         * Figures out the new setPoint
043         * @param newValue The new value to add to the current input to find the setPoint
044         * @return the new setPoint
045         */
046        protected double computeSetpoint(double newValue) {
047            return returnPIDInput()+newValue;
048        }
049    }
