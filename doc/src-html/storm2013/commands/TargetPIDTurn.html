001    package storm2013.commands;
002    
003    import storm2013.Robot;
004    import storm2013.utilities.Target;
005    
006    /** 
007     * Does horizontal target tracking (turning the robot). It reads its target
008     * angle from the camera, then uses the gyro as feedback to reach that angle.
009     */
010    public class TargetPIDTurn extends TargetPIDCommand {    
011        /** see {@link TargetPIDCommand} */
012        public TargetPIDTurn(Target target, double timeout, boolean continuous){
013            super(target,Target.Axis.X,timeout,continuous,
014                  0.5, 0.1, 1, // P, I, D
015                  4,0.7,       // tolerance, maxOutput (higher because of carpet)
016                  1.0/20);     // period
017            requires(Robot.driveTrain);
018        }
019        
020        protected double returnPIDInput() {
021            return Robot.driveTrain.getGyroAngle();
022        }
023        protected void writePIDOut(double output) {
024            Robot.driveTrain.tankDrive(output, -output);
025        }
026        protected void useCameraValue(double value) {
027            setSetpoint(returnPIDInput()+value);
028        }
029    }
