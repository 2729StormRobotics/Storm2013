001    package storm2013.commands;
002    
003    import storm2013.Robot;
004    import storm2013.subsystems.Vision.Distance;
005    import storm2013.utilities.StringPot;
006    
007    public class TiltForDistance extends StringPotTilt {
008        private static double _targetFromDistance(Distance distance) {
009            if(distance == Distance.NEAR) {
010                return StringPot.VAL_NEAR;
011            } else if(distance == Distance.CENTER) {
012                return StringPot.VAL_CENTER;
013            } else if(distance == Distance.OPPONENT_AUTO) {
014                return StringPot.VAL_OPPAUTO;
015            } else if(distance == Distance.FEEDER) {
016                return StringPot.VAL_FEEDER;
017            }
018            return -1;
019        }
020        private Distance _distance;
021        public TiltForDistance(double power,Distance distance) {
022            super(power, _targetFromDistance(distance));
023            _distance = distance;
024        }
025        
026        protected Distance _getDistance() {
027            return _distance;
028        }
029        protected void _setDistance(Distance distance) {
030            _distance = distance;
031            _setTarget(_targetFromDistance(distance));
032        }
033    
034        protected void initialize() {
035            super.initialize();
036        }
037    
038        protected boolean isFinished() {
039            return Robot.oi.getTilterAxis() != 0 || super.isFinished();
040        }
041    }
