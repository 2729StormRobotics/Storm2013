001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /**
007     * Prints out values based on teleoperated movement that can be used for an
008     * autonomouse.
009     * @author evan1026
010     */
011    public class PrintAutonomousMove extends Command {
012    
013        private double _speed;
014        private double _waitTime;
015        
016        /**
017         * Constructor and such.
018         * @param speed The speed to move at when doing these autonomously.
019         * @param waitTime The amount of time to wait in between different movement commands.
020         */
021        public PrintAutonomousMove(double speed,double waitTime){
022            _speed = speed;
023            _waitTime = waitTime;
024        }
025        
026        /**
027         * Initializes the {@link Command}. This actually does all of the logic, and
028         * execute does nothing.
029         */
030        protected void initialize() {
031            String out = "addSequential(new ";
032            double left = Robot.driveTrain.getLeftDistance();
033            double right = Robot.driveTrain.getRightDistance();
034            double turn = Robot.driveTrain.getGyroAngle();
035            
036            if ((left < 0) != (right < 0)){
037                out += "GyroTurn(";
038                out += turn;
039            }
040            else {
041                out += "EncoderDrive(";
042                out += (right < 0) ? "-" : "";
043                out += (Math.min(Math.abs(left), Math.abs(right)));
044            }
045            
046            out += ", " + _speed + "));";
047            System.out.println(out);
048            
049            System.out.println("addSequential(new DoNothing()," + _waitTime + ");");
050            
051            Robot.driveTrain.clearEncoder();
052        }
053    
054        /**
055         * Does nothing.
056         */
057        protected void execute() {
058        }
059    
060        /**
061         * Tells the robot to end the {@link Command}.
062         * @return true, so the {@link Command} ends immediately.
063         */
064        protected boolean isFinished() {
065            return true;
066        }
067    
068        /**
069         * Does nothing.
070         */
071        protected void end() {
072        }
073    
074        /**
075         * Calls {@link #end()}
076         */
077        protected void interrupted() {
078            end();
079        }
080        
081    }
