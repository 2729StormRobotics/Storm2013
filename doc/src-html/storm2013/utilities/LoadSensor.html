001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.AnalogChannel;
004    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
005    import edu.wpi.first.wpilibj.tables.ITable;
006    
007    /**
008     * Provides the ability to easily interface with the load sensor.
009     * @author Storm
010     */
011    public class LoadSensor implements LiveWindowSendable{
012        
013        AnalogChannel loadSensor;
014        
015        ITable _table;
016        
017        //value is slope: x-axis: amps y-axis: volts (Dropbox/Storm2729/current-sensor-curve.xlsx)
018        final double voltToAmpRatio = 39.46e-3;
019        
020        /**
021         * 
022         * @param channel The analog channel where the load sensor is plugged in.
023         */
024        public LoadSensor(int channel){
025            loadSensor = new AnalogChannel(channel);
026        }
027        
028        /**
029         * Gets how many amps are flowing through the sensor. May be a bit inaccurate,
030         * since it's calculated based on the voltage.
031         * @return The value in amps
032         */
033        public double getAmps(){
034            return loadSensor.getVoltage() / voltToAmpRatio;
035        }
036        
037        /**
038         * Gets how many volts the load sensor is reading.
039         * @return The value in volts
040         */
041        public double getVolts(){
042            return loadSensor.getVoltage();
043        }
044        
045        /**
046         * Puts the numbers into the network table
047         */
048        public void updateTable() {
049            if(_table != null) {
050                _table.putNumber("voltage", loadSensor.getVoltage());
051                _table.putNumber("amperage", getAmps());
052            }
053        }
054    
055        /**
056         * LiveWindow wants it. I don't really know what it is supposed to do. It does
057         * nothing anyway.
058         */
059        public void startLiveWindowMode() {}
060    
061        /**
062         * LiveWindow wants it. I don't really know what it is supposed to do. It does
063         * nothing anyway.
064         */
065        public void stopLiveWindowMode() {}
066    
067        /**
068         * Sets up the network table for this to write to
069         * @param table The network table for this to write to
070         */
071        public void initTable(ITable table) {
072            _table = table;
073            updateTable();
074        }
075    
076        /**
077         * Returns the network table this is writing to.
078         * @return The table
079         */
080        public ITable getTable() {
081            return _table;
082        }
083    
084        /**
085         * Returns the type to be used for smart dashboard. I can't say I fully know
086         * what it does, but if someone does, they can change this documentation.
087         * @return "LoadSensorType"
088         */
089        public String getSmartDashboardType() {
090            return "LoadSensorType";
091        }
092    }
