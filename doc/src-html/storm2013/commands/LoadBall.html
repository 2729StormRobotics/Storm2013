001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.DigitalInput;
004    import edu.wpi.first.wpilibj.command.Command;
005    import storm2013.RobotMap;
006    
007    /**
008     * Moves a ball up the back to get it ready to be fired. All this does is create 
009     * a new {@link ControlLoader}, then it an heros.
010     * @author evan1026
011     */
012    public class LoadBall extends Command {
013    
014        private DigitalInput ballLoadedSensor;
015        private ControlLoader controlLoader;
016        
017        /**
018         * Creates a new instance of the command.
019         */
020        public LoadBall(){
021            ballLoadedSensor = new DigitalInput(RobotMap.PORT_BREAKBEAM_BALL_READY);
022            controlLoader = new ControlLoader(true);
023        }
024        
025        /**
026         * {@inheritDoc}
027         */
028        protected void initialize() {
029        }
030    
031        /**
032         * {@inheritDoc}
033         */
034        protected void execute() {
035        }
036    
037        /**
038         * {@inheritDoc}
039         * @return {@inheritDoc}
040         */
041        protected boolean isFinished() {
042            return !ballLoadedSensor.get();
043        }
044    
045        /**
046         * {@inheritDoc}
047         */
048        protected void end() {
049            controlLoader.end();
050        }
051        
052       /**
053        * {@inheritDoc}
054        */
055        protected void interrupted() {
056            end();
057        }
058        
059    }
