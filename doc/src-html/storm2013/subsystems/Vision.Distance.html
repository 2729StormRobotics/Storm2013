001    package storm2013.subsystems;
002    
003    import edu.wpi.first.wpilibj.command.Subsystem;
004    import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
005    import storm2013.commands.StringPotTilt;
006    import storm2013.commands.TiltForDistance;
007    
008    /**
009     * Subsystem providing an interface with StormCV to control the crosshair
010     * location.
011     */
012    public class Vision extends Subsystem {
013        /**
014         * These are the distances we've calibrated crosshair locations for.
015         */
016        public static final class Distance {
017            /** From the back of the pyramid forward. */
018            public static final Distance NEAR          = new Distance("Near");
019            /** Between the auto lines (at an angle). */
020            public static final Distance CENTER        = new Distance("Center line");
021            /** At the opponent auto line (at an angle). */
022            public static final Distance OPPONENT_AUTO = new Distance("Opponent Auto");
023            /** From the feeder station. */
024            public static final Distance FEEDER        = new Distance("Feeder");
025            
026            private String _value;
027            
028            private Distance(String value) {
029                _value = value;
030            }
031            
032            public String getValue() {
033                return _value;
034            }
035        }
036        
037        private static final Distance[] _distances = {
038            Distance.NEAR,
039            Distance.CENTER,
040            Distance.OPPONENT_AUTO,
041            Distance.FEEDER
042        };
043        
044        private int _currIndex = 0;
045        
046        public Vision() {
047            _setDistance(0);
048        }
049        
050    
051        protected void initDefaultCommand() {}
052        
053        /** Set distance (manually, no cycling). */
054        public void setDistance(Distance distance) {
055            for(int i=0;i<_distances.length;++i) {
056                if(_distances[i] == distance) {
057                    _setDistance(i);
058                    return;
059                }
060            }
061        }
062        
063        private void _setDistance(int index) {
064            index = index%_distances.length;
065            if(index < 0) {
066                index += _distances.length;
067            }
068            _currIndex = index;
069            SmartDashboard.putString("Distance", _distances[index].getValue());
070        }
071        
072        /** Cycle through distances forward (near -> feeder). */
073        public Distance nextDistance() {
074            _setDistance(_currIndex+1);
075            return _distances[_currIndex];
076        }
077        
078        /** Cycle through distances backward (feeder -> near). */
079        public Distance prevDistance() {
080            _setDistance(_currIndex-1);
081            return _distances[_currIndex];
082        }
083    
084        public Distance getCurrDistance() {
085            return _distances[_currIndex];
086        }
087    }
