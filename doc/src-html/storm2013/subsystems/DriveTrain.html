005    package storm2013.subsystems;
006    
007    import edu.wpi.first.wpilibj.RobotDrive;
008    import edu.wpi.first.wpilibj.Victor;
009    import edu.wpi.first.wpilibj.command.Subsystem;
010    import edu.wpi.first.wpilibj.livewindow.LiveWindow;
011    import storm2013.RobotMap;
012    import storm2013.commands.ArcadeDrive;
013    import storm2013.utilities.AccelerationLimiter;
014    
015    /**
016     *
017     * @author Joe
018     */
019    public class DriveTrain extends Subsystem {
020        
021        private double ACCEL_RATE =1.7;
022        private double DECEL_RATE = 1.7;
023        
024        Victor _left = new Victor(RobotMap.PORT_MOTOR_DRIVE_LEFT),_right = new Victor(RobotMap.PORT_MOTOR_DRIVE_RIGHT);
025        
026        AccelerationLimiter _leftControl = new AccelerationLimiter(_left, ACCEL_RATE, DECEL_RATE);
027        AccelerationLimiter _rightControl = new AccelerationLimiter(_right, ACCEL_RATE, DECEL_RATE);
028        RobotDrive _drive = new RobotDrive(_leftControl,_rightControl);
029        
030    
031        public DriveTrain() {
032            LiveWindow.addActuator("Drive Train", "Left Motor", _left);
033            LiveWindow.addActuator("Drive Train", "Right Motor", _right);
034        }
035        
036        public void initDefaultCommand() {
037            setDefaultCommand(new ArcadeDrive());
038        }
039        
040        public void arcadeDrive(double speed,double turn) {
041            _drive.arcadeDrive(-speed, turn);
042        }
043        
044        public void tankDrive(double leftValue, double rightValue){
045            _drive.tankDrive(leftValue, rightValue);
046        }
047    }
