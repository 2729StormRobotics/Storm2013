001    package storm2013.commands.LEDcommands;
002    
003    import edu.wpi.first.wpilibj.Timer;
004    import edu.wpi.first.wpilibj.command.Command;
005    import storm2013.Robot;
006    
007    /** Flash through a series of colors */
008    public class Flash extends Command {
009        public static final double DEFAULT_PERIOD = 1; //once a second
010        
011        protected Timer _timer = new Timer();
012        protected Color[] _colors;
013        protected double  _period;
014        protected int index;
015    
016        public Flash(Color[] colors) {
017            this(colors, DEFAULT_PERIOD);
018        }
019        
020        /**
021         * @param colors Sequence of colors
022         * @param period Time each color is on, in seconds
023         */
024        public Flash(Color[] colors, double period) {
025            _colors = colors;
026            _period = period;
027            requires(Robot.ledStrip);
028        }
029        
030        protected void initialize() {
031            _timer.start();
032        }
033        protected void execute() {
034            double time = _timer.get();
035            
036            if(time >= _period){
037                Robot.ledStrip.setColor(_colors[index].getRed(), _colors[index].getGreen(), _colors[index].getBlue());
038                index = (index + 1) % _colors.length;
039                _timer.reset();
040            }
041        }
042        protected boolean isFinished() {
043            return false;
044        }
045        protected void end() {
046            _timer.stop();
047        }
048        protected void interrupted() {
049            end();
050        }
051        
052        protected void setPeriod(double period){
053            _period = period;
054        }
055        protected double getPeriod(){
056            return _period;
057        }
058    }
