001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /** Moves the tilter upward. */
007    public class RaiseTilter extends Command {
008        public RaiseTilter() {
009            requires(Robot.tilter);
010        }
011    
012        protected void initialize() {
013            System.out.println("Tilting");
014            Robot.tilter.moveUp();
015        }
016        protected void execute() {}
017        protected boolean isFinished() {
018            return false;
019        }
020        protected void end() {
021            Robot.tilter.stop();
022        }
023        protected void interrupted() {
024            end();
025        }
026    }
