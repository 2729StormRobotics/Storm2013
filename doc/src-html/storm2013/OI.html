085        private double _zeroDeadzone(double joyValue,double dead) {
086            return Math.abs(joyValue) > dead ? joyValue : 0;
087        }
088        
089        private double _applyExponential(double joyValue) {
090            if(joyValue == 0) {
091                return 0;
092            }
093            double sign = (joyValue > 0) ? 1 : -1;
094            return sign*joyValue*joyValue/Math.sqrt(Math.abs(joyValue));
095        }
096        
097        public double getLeftDrive() {
098            return _zeroDeadzone(-driveJoystick.getRawAxis(RobotMap.AXIS_DRIVE_LEFT),0.15)*(slowModeButton.get() ? 0.5 : 1);
099        }
100        
101        public double getRightDrive() {
102            return _zeroDeadzone(-driveJoystick.getRawAxis(RobotMap.AXIS_DRIVE_RIGHT),0.15)*(slowModeButton.get() ? 0.5 : 1);
103        }
104        
105        public double getTilterAxis() {
106            return -driveJoystick.getRawAxis(RobotMap.AXIS_TILTER);
107        }
