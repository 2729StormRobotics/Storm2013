001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    
006    /**
007     * Raises the tilter
008     * @author Joe
009     */
010    public class RaiseTilter extends Command {
011        
012        /**
013         * Constructor.
014         */
015        public RaiseTilter() {
016            requires(Robot.tilter);
017        }
018    
019        /**
020         * Initializes the {@link Command}.
021         */
022        protected void initialize() {
023            System.out.println("Tilting");
024            Robot.tilter.moveUp();
025        }
026    
027        /**
028         * Does nothing
029         */
030        protected void execute() {}
031    
032        /**
033         * Returns false to make it run while a button is pressed.
034         * @return false
035         */
036        protected boolean isFinished() {
037            return false;
038        }
039    
040        /**
041         * Stops the tilter.
042         */
043        protected void end() {
044            Robot.tilter.stop();
045        }
046    
047        /**
048         * Calls {@link #end()}
049         */
050        protected void interrupted() {
051            end();
052        }
053    }
