001    package storm2013.commands;
002    
003    import storm2013.Robot;
004    import storm2013.utilities.Target;
005    
006    /**
007     * Turns the robot based on camera values and values from the gyro
008     * @author evan1026
009     */
010    public class TargetPIDTurn extends TargetPIDCommand {    
011        /**
012         * It's a constructor
013         * @param target The {@link Target} at which we're aiming
014         * @param timeout How long (seconds) before the command times out
015         * @param continuous True if the command will actually finish (stop running) and false otherwise
016         */
017        public TargetPIDTurn(Target target, double timeout, boolean continuous){
018            super(target,Target.Axis.X,timeout,continuous,
019                  0.01, 0, 0.1, // P, I, D
020                  2,0.6);
021            requires(Robot.driveTrain);
022        }
023        
024        /**
025         * Gets the angle from the gyro to be used for the PID controller
026         * @return the gyro angle
027         */
028        protected double returnPIDInput() {
029            return Robot.driveTrain.getGyroAngle();
030        }
031    
032        /**
033         * Uses the value from the PID controller to turn the robot.
034         * @param output the value to use (between -6 and 6)
035         */
036        protected void writePIDOut(double output) {
037            Robot.driveTrain.tankDrive(output, -output);
038            System.out.println("Turning @ " + output + " (" + getPIDController().getP() + "," + getPIDController().getI() + "," + getPIDController().getD());
039        }
040    }
