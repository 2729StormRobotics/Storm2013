001    package storm2013.commands.LEDcommands;
002    
003    import edu.wpi.first.wpilibj.DriverStation;
004    import edu.wpi.first.wpilibj.DriverStation.Alliance;
005    import edu.wpi.first.wpilibj.Timer;
006    import edu.wpi.first.wpilibj.command.Command;
007    import storm2013.Robot;
008    
009    /**
010     * Turns lights between white and off, in a sinusoid, with frequency
011     * proportional to speed.
012     */
013    public class SetModeMoving extends Command {
014        private static final double PERIOD_SCALAR = 0.1;
015        
016        private Timer _timer = new Timer();
017        private double _val = 0;
018        
019        public SetModeMoving(){
020            requires(Robot.ledStrip);
021        }
022        
023        public void execute() {
024            Alliance alliance = DriverStation.getInstance().getAlliance();
025            double r = 0,g = 0,b = 0;
026            if(alliance.value == Alliance.kRed_val) {
027                r = 1;
028                g = b = 0;
029            } else if(alliance.value == Alliance.kBlue_val) {
030                b = 1;
031                r = g = 0;
032            } else {
033                r = g = b = 1;
034            }
035            double rate = 2*Math.PI*(Math.abs(Robot.driveTrain.getLeftSpeed()) + Math.abs(Robot.driveTrain.getRightSpeed())) / 2;
036            if(rate == 0) {
037                _val = Math.PI/2;
038            } else {
039                _val += rate*_timer.get()/PERIOD_SCALAR;
040                _val %= Math.PI*2;
041            }
042            _timer.reset();
043            double scalar = (Math.sin(_val)+1)/2;
044            if(scalar <= 0.25) {
045                scalar = 0;
046            } else {
047                scalar = (scalar-0.25)/0.75;
048            }
049            double color = scalar*255;
050            Robot.ledStrip.setColor((int)(r*color),(int)(g*color),(int)(b*color));
051        }
052        protected void initialize() {
053            _timer.reset();
054            _timer.start();
055        }
056        protected boolean isFinished() {
057            return false;
058        }
059        protected void end() {
060            _timer.stop();
061        }
062        protected void interrupted() {
063            end();
064        }
065    }
