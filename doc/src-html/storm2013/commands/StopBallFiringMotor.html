028         */

031        }
032    
033        /**
034         * {@inheritDoc}
035         */
036        protected void execute() {
037        }
038

040         * {@inheritDoc}
041         */

044        }
045    
046        /**
047         * {@inheritDoc}
048         */
049        protected void end() {
050        }
051    
052        /**
053         * {@inheritDoc}
054         */
055        protected void interrupted() {
056            end();
057        } 
058    }
