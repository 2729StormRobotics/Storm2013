001    package storm2013.commands.LEDcommands;
002    
003    import storm2013.Robot;
004    
005    /**
006     * Flashes the LEDs, getting faster as the wheel approaches the target speed.
007     */
008    public class SetModeSpinningUp extends FlashOnAndOff {
009        public SetModeSpinningUp(){
010            super(Color.SPINNING_UP);
011        }
012        
013        private static final double PERIOD_SCALAR = 1.0/2500;
014    
015        public void execute() {
016            setPeriod(Math.abs(Robot.shooter.getPIDController().getError())*PERIOD_SCALAR);
017            super.execute();
018        }
019    }
