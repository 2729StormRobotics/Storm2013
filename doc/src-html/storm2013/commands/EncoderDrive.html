001    package storm2013.commands;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import storm2013.Robot;
005    import storm2013.subsystems.DriveTrain;
006    
007    /**
008     * Drives a certain number of wheel rotations based on an encoder. The encoder values
009     * are supplied by the drive train (see {@link Robot}), which is implemented by some other class
010     * to supply values in rotations. It then drives at the specified drive speed until
011     * until it reaches the goal.
012     * 
013     * @author evan1026
014     */
015    public class EncoderDrive extends Command {
016    
017        private double _goal;
018        private double _dist;
019        private double _driveSpeed;
020        private DriveTrain _driveTrain = Robot.driveTrain;
021        
022        /**
023         * Main constructor for the class. More could be created if it proves to be worthwhile,
024         * but for now, use this one.
025         * 
026         * @param goal            How far you want to go (use rotations for units, just to make a standard).
027         * @param driveSpeed      The speed at which you want to travel (also in rotations).
028         */
029        public EncoderDrive(double goal, double driveSpeed){ 
030            
031            _goal = goal;
032            _driveSpeed = driveSpeed;
033            requires(Robot.driveTrain);
034        }
035        
036        /**
037         * Implemented from {@link Command}, but it doesn't actually do anything at the moment.
038         */
039         protected void initialize() {
040            Robot.driveTrain.clearEncoder();
041         }
042        
043        /**
044         * Implemented from {@link Command}. It is run over and over and such, and it calculates
045         * what the drive speed the robot needs to accelerate toward by setting values in 
046         * the {@link DriveTrain}.
047         * 
048         * @see DriveTrain
049         */
050        public void execute() {
051            double left  = Robot.driveTrain.getLeftDistance(),
052                   right = Robot.driveTrain.getRightDistance();
053            
054            if(Math.abs(_goal-left) > Math.abs(_goal-right)) {
055                _dist = left;
056            }
057            else {
058                _dist = right;
059            }
060            
061            
062            if (_goal > 0) {
063                _driveTrain.tankDrive(_driveSpeed, _driveSpeed);
064            }
065            else {
066                _driveTrain.tankDrive(-_driveSpeed, -_driveSpeed);
067            }
068        }
069        
070        /**
071         * Implemented from {@link Command}. When it returns true, it runs {@link #end()},
072         * and then stops running the command.
073         * 
074         * @return true if it has reached the goal and false otherwise.
075         */
076        public boolean isFinished() {
077            if(_goal < 0) {
078                return _dist <= _goal;
079            } 
080            else {
081                return _dist >= _goal;
082            }
083        }
084        
085        /**
086         * Implemented from {@link Command}. This is the last thing run before the command
087         * stops running. Stops the robot.
088         */
089        public void end() {
090            Robot.driveTrain.tankDrive(0, 0);
091        }
092        
093        /**
094         * Implemented from {@link Command}. When {@link Command#cancel()} is run, 
095         * this is first called, then {@link #end()}.
096         */
097        public void interrupted() {
098            end();
099        }
100    
101        /**
102         * Gets the current goal.
103         * @return The current goal, in rotations.
104         */
105        public double getGoal() {
106            return _goal;
107        }
108        
109        /**
110         * Sets a new goal.
111         * @param goal The new goal, in rotations.
112         */
113        public void setGoal(double goal) {
114            _goal = goal;
115        }
116        
117        /**
118         * Gets the total distance traveled.
119         * @return The distance traveled, in revolutions.
120         */
121        public double getDist() {
122            return _dist;
123        }
124       
125        /**
126         * Returns the speed at which the robot it set to drive.
127         * @return The drive speed, in RPM.
128         */
129        public double getDriveSpeed() {
130            return _driveSpeed;
131        }
132        
133        /**
134         * Sets the drive speed.
135         * @param driveSpeed The drive speed, in RPM.
136         */
137        public void setDriveSpeed(double driveSpeed) {
138            _driveSpeed = driveSpeed;
139        }
140    }
