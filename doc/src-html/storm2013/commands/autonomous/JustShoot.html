001    package storm2013.commands.autonomous;
002    
003    import edu.wpi.first.wpilibj.command.Command;
004    import edu.wpi.first.wpilibj.command.CommandGroup;
005    import storm2013.Robot;
006    import storm2013.commands.LowerTilter;
007    import storm2013.commands.Shoot;
008    import storm2013.commands.SpinDown;
009    import storm2013.commands.TargetPIDTilt;
010    import storm2013.subsystems.Vision;
011    import storm2013.utilities.Target;
012    
013    /**
014     * This autonomous mode just shoots a full clip; we use it from either back
015     * corner of the pyramid.
016     */
017    public class JustShoot extends CommandGroup {
018        public JustShoot() {
019            // Make sure the vision system is targetting for the right part of the
020            // field
021            addParallel(new Command() {
022                {
023                    requires(Robot.vision);
024                }
025                protected void initialize() {
026                    Robot.vision.setDistance(Vision.Distance.NEAR);
027                }
028                protected void execute() {}
029                protected boolean isFinished() {
030                    return true;
031                }
032                protected void end() {}
033                protected void interrupted() {}
034            });
035            // Lower the tilter so the target is in sight
036            addSequential(new LowerTilter(),1);
037            // Align the tilter with the target
038            addSequential(new TargetPIDTilt(Target.ThreePT, 1.0, false));
039            // Shoot repeatedly (in case of jams)
040            addSequential(new Shoot());
041            addSequential(new Shoot());
042            addSequential(new Shoot());
043            addSequential(new Shoot());
044            addSequential(new Shoot());
045            addSequential(new Shoot());
046            // Stop shooting (not entirely necessary -- disabled mode would spin it
047            // down anyway)
048            addSequential(new SpinDown());
049        }
050    }
