001    package storm2013.utilities;
002    
003    import edu.wpi.first.wpilibj.PWM;
004    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
005    import edu.wpi.first.wpilibj.tables.ITable;
006    import edu.wpi.first.wpilibj.tables.ITableListener;
007    
008    /**
009     * A class that makes it easier to deal with LED's.
010     * @author Joe
011     */
012    public class LED implements LiveWindowSendable {
013        private PWM _led;
014        private ITable _table;
015        
016        /**
017         * 
018         * @param port The port the LED is plugged into
019         */
020        public LED(int port) {
021            _led = new PWM(port);
022        }
023        
024        /**
025         * Sets the value for the LED.
026         * @param value The value to set between 0-255.
027         */
028        public void set(int value) {
029            _led.setRaw(value);
030        }
031        
032        /**
033         * Gets the value being output to the LED
034         * @return The value being output
035         */
036        public int get() {
037            return _led.getRaw();
038        }
039    
040        /**
041         * Puts the LED output value into the network table.
042         */
043        public void updateTable() {
044            if(_table != null) {
045                _table.putNumber("Value", get()*2/255.0-1);
046            }
047        }
048    
049        /**
050         * Sets stuff up so this all works with LiveWindow.
051         */
052        public void startLiveWindowMode() {
053            _led.setRaw(0);
054            ITableListener table_listener = new ITableListener() {
055                public void valueChanged(ITable itable, String key, Object value, boolean bln) {
056                    set(Math.min((int)((((Double) value).doubleValue()+1)*255.0/2),255));
057                }
058            };
059            _table.addTableListener("Value", table_listener, true);
060        }
061    
062        /**
063         * Doesn't do anything.
064         */
065        public void stopLiveWindowMode() {
066        }
067    
068        /**
069         * Initializes the table the LED writes to.
070         * @param subtable The table to write to
071         */
072        public void initTable(ITable subtable) {
073            _table = subtable;
074            updateTable();
075        }
076    
077        /**
078         * Gets the table this is writing to.
079         * @return That table
080         */
081        public ITable getTable() {
082            return _table;
083        }
084    
085        /**
086         * Returns the type to be used for smart dashboard. I can't say I fully know
087         * what it does, but if someone does, they can change this documentation.
088         * @return "Speed Controller"
089         */
090        public String getSmartDashboardType() {
091            return "Speed Controller";
092        }
093    }
