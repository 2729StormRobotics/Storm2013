005    package storm2013.utilities;
006    
007    import edu.wpi.first.wpilibj.PIDOutput;
008    import edu.wpi.first.wpilibj.livewindow.LiveWindowSendable;
009    import edu.wpi.first.wpilibj.parsing.IUtility;
010    import edu.wpi.first.wpilibj.tables.ITable;
011    import edu.wpi.first.wpilibj.tables.ITableListener;
012    import java.util.Timer;
013    import java.util.TimerTask;
014    
015    
016    /**
017     *
018     * @author evan1026
019     */
020    
021    public class BangBangController implements IUtility, LiveWindowSendable {
022        
023        private PIDOutput     _PIDOut;
024        private IEncoderSpeed _SpeedCalculator;
025        private double        _setPoint;
026        private double        _spinUpPoint;
027        private double        _lastSpeed;
028        private double        _spinUpSpeed;
029        private double        _maxSpeed;
030        private Timer         _timer;
031        private ITable        _table;
032        
033        private ITableListener _listener = new ITableListener(){
034            public void valueChanged(ITable itable, String key, Object value, boolean isNew) {
035                if (key.equals("setPoint")) {
036                    setSetPoint(_table.getNumber("setPoint"));
037                }
038                else if (key.equals("spinUpPoint")) {
039                    setSpinUpPoint(_table.getNumber("spinUpPoint"));
040                }
041                else if (key.equals("spinUpSpeed")) {
042                    setSpinUpSpeed(_table.getNumber("spinUpSpeed"));
043                }
044                else if (key.equals("maxSpeed")) {
045                    setMaxSpeed   (_table.getNumber("maxSpeed"));
046                }
047            }
048        };
049    
050        public BangBangController(PIDOutput PIDOut, IEncoderSpeed SpeedCalculator, ITable table,
051                double setPoint, double spinUpPoint, double spinUpSpeed, double maxSpeed, 
052                double period){
053            
054            _PIDOut          = PIDOut;
055            _SpeedCalculator = SpeedCalculator;
056            _setPoint        = setPoint;
057            _spinUpPoint     = spinUpPoint;
058            _spinUpSpeed     = spinUpSpeed;
059            _maxSpeed        = maxSpeed;
060            _timer           = new Timer();
061            
062            _timer.schedule(new BangBangTask(this), 0L, (long)(period * 1000));
063            
064            //InitTable(table);
065        }
066    
067        public void go() {
068            double currSpeed = _SpeedCalculator.getRPM();
069            if (!(currSpeed == _lastSpeed)){
070                if (Math.abs(currSpeed) < Math.abs(_spinUpPoint)){
071                    _PIDOut.pidWrite(_spinUpSpeed);
072                }
073                else if (Math.abs(currSpeed) < Math.abs(_setPoint)){
074                    _PIDOut.pidWrite(_maxSpeed);
075                }
076                else{
077                    _PIDOut.pidWrite(0);
078                }
079                _lastSpeed = currSpeed;
080            }
081        }
082        
083        public void initTable(ITable table){
084            if(_table != null) {
085                _table.removeTableListener(_listener);
086            }
087            _table = table;
088            
089            if (table != null){
090                _table.putNumber("setPoint",    _setPoint);
091                _table.putNumber("spinUpPoint", _spinUpPoint);
092                _table.putNumber("spinUpSpeed", _spinUpSpeed);
093                _table.putNumber("maxSpeed",    _maxSpeed);
094                _table.addTableListener(_listener, false);
095            }
096        }
097    
098        /*Gets and sets*/
099        public double getSetPoint() {
100            return _setPoint;
101        }
102        public void setSetPoint(double setPoint) {
103            _setPoint = setPoint;
104            _table.putNumber("setPoint", _setPoint);
105        }
106        public double getSpinUpPoint() {
107            return _spinUpPoint;
108        }
109        public void setSpinUpPoint(double spinUpPoint) {
110            _spinUpPoint = spinUpPoint;
111            _table.putNumber("spinUpPoint", _spinUpPoint);
112        }
113        public double getSpinUpSpeed() {
114            return _spinUpSpeed;
115        }
116        public void setSpinUpSpeed(double spinUpSpeed) {
117            _spinUpSpeed = spinUpSpeed;
118            _table.putNumber("spinUpSpeed", _spinUpSpeed);
119        }
120        public double getMaxSpeed() {
121            return _maxSpeed;
122        }
123        public void setMaxSpeed(double maxSpeed) {
124            _maxSpeed = maxSpeed;
125            _table.putNumber("maxSpeed", _maxSpeed);
126        }
127        public ITable getTable(){
128            return _table;
129        }
130        
131        public String getSmartDashboardType(){
132            return "BangBangController";
133        }
134    
135        public void updateTable() {
136        }
137    
138        public void startLiveWindowMode() {
139        }
140    
141        public void stopLiveWindowMode() {
142        }
143        
144        private class BangBangTask extends TimerTask{
145    
146            private BangBangController _controller;
147            
148            public BangBangTask(BangBangController controller) {
149                if (controller == null) {
150                    throw new NullPointerException("Given PIDController was null");
151                }
152                _controller = controller;
153            }
154            
155            public void run() {
156                _controller.go();
157            }
158        
159        }
160    }
